\begin{definition}
     Define \textbf{summation}, or $\sum_{i=i_0}^n a_i$ for $n \in \mathbb{Z}^+$ and $n \geq i_0$ as follows: \[
        \sum_{i=i_0}^{i_0} = a_{i_0} \qquad \sum_{i=i_0}^{n}a_i=a_n+\sum_{i=i_0}^{n-1}  a_i
     \]
\end{definition}
\begin{definition}
     Define \textbf{product}, or $\prod_{i=i_0}^n a_i$ for $n \in \mathbb{Z}^+$ and $n \geq i_0$ as follows: \[
        \prod_{i=i_0}^{i_0} = a_{i_0} \qquad \prod_{i=i_0}^{n}a_i=a_n \cdot \prod_{i=i_0}^{n-1}  a_i
     \]
\end{definition}
For properties of products and summations, see Appendix A.
