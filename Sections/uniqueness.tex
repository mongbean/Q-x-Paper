\begin{lemma}
    If $f(x) \in \qx$ can be expressed as $u(x)\prod _{i=1}^n p_i(x)^{\alpha_i}$ where $p_i(x) \in \qx$ is prime and unique, $u(x) \in \qx^{\times}$,
    $\alpha_i \in \bZ^+$, then only the set $S= \{(p_i, \alpha_i)\}$ has this property. 
\end{lemma}
\begin{proof}
For the sake of contradiction, there exists $A = \{(q_i, \beta_i)\}$ of size $m$, s.t. \[f(x) = u(x)\prod _{i=1}^mq_i(x)^{\beta_i}\] and $\exists (q_k, \beta_k) \in A$ s.t. $(q_k, \beta_k) \notin S$. This can be true in two ways.\\
\textbf{Case 1: } $\forall \phi \in \pZ$, $(q_k(x), \phi) \notin S$.  
\begin{align*}
    u(x)\prod _{i=1}^mq_i(x)^{\beta_i} &= u(x)\prod _{i=1}^n p_i(x)^{\alpha_i} \\
    \prod _{i=1}^mq_i(x)^{\beta_i} &= \prod _{i=1}^n p_i(x)^{\alpha_i}\\
    \prod _{i=1}^{k-1}q_i(x)^{\beta_i} \m q_k^{\beta_k} \m \prod _{i=k+1}^{m}q_i(x)^{\beta_i} &= \prod _{i=1}^n p_i(x)^{\alpha_i}\\
\end{align*}
Thus, we know that $q_k(x)^{\beta_k} \mid \prod _{i=1}^n p_i(x)^{\alpha_i}$. By Lemma ???? we know that $q_k^{\beta_k} \mid p_i(x)^{\alpha_i}$ for some $i$, yet since $p_i \neq q_k$, we know that 
\[\forall p_i \in S \quad \gcd(p_i^{\alpha_i}, q_k^{\beta_k}) = 1 \qquad \text{Lemma ??????}\] Thus, this is a contradiction.\\
\textbf{Case 2: } $\forall  a(x) \in \qx$, $(a(x), \beta_k) \notin S$. From the above, we know that $\exists (p_j(x), \alpha_j) \in S$ such that $p_j(x) = q_k(x)$. Again, 
\begin{align*}
    u(x)\prod _{i=1}^mq_i(x)^{\beta_i} &= u(x)\prod _{i=1}^n p_i(x)^{\alpha_i} \\
    \prod _{i=1}^mq_i(x)^{\beta_i} &= \prod _{i=1}^n p_i(x)^{\alpha_i}\\
    \prod _{i=1}^{k-1}q_i(x)^{\beta_i} \m q_k^{\beta_k} \m \prod _{i=k+1}^{m}q_i(x)^{\beta_i} &= \prod _{i=1}^{j-1}p_i(x)^{\alpha_i} \m q_j^{\alpha_j} \m \prod _{i=j+1}^{n}p_i(x)^{\alpha_i}\\
\end{align*}
Now, we have 2 cases, as $\beta_k \neq \alpha_j$. \\
\textbf{Case 2.1: }$\beta_k > \alpha_j$ 
\begin{align*}
    \prod _{i=1}^{k-1}q_i(x)^{\beta_i} \m q_k^{\beta_k} \m \prod _{i=k+1}^{m}q_i(x)^{\beta_i} &= \prod _{i=1}^{j-1}p_i(x)^{\alpha_i} \m q_j^{\alpha_j} \m \prod _{i=j+1}^{n}p_i(x)^{\alpha_i}\\
    \prod _{i=1}^{k-1}q_i(x)^{\beta_i} \m q_k^{\beta_k-\alpha_j} \m \prod _{i=k+1}^{m}q_i(x)^{\beta_i} &= \prod _{i=1}^{j-1}p_i(x)^{\alpha_i} \m \prod _{i=j+1}^{n}p_i(x)^{\alpha_i}\\
\end{align*}
Thus, $q_k^{\beta_k-\alpha_j} \mid \prod _{i=1}^{j-1}p_i(x)^{\alpha_i}$ or $q_k^{\beta_k-\alpha_j} \mid \prod _{i=j+1}^{n}p_i(x)^{\alpha_i}$ by the Fundamental Lemma. However, neither can be true as $\forall p_i \in S \quad \gcd(p_i^{\alpha_i}, q_k^{\beta_k}) = 1$. Thus, this is a contradiction.\\
\textbf{Case 2.2: }$\beta_k <\alpha_j$ can be proven using an analogous argument. 
\end{proof}

\begin{lemma}
    $ \gcd(p^k, q^j) = 1$ for all $p, q \in \qx$ such that $\gcd(p, q) = 1$.
\end{lemma}
\begin{proof}
    First, note that if $k=0$ or $j=0$, $\gcd(f(x), 1) = 1 \forall f(x)\in \qx$.
    Next, prove that $\gcd(p, q^j) = 1$. Let $S=\left\{i \in \pZ\mid \gcd(p, q^i) \neq 1 \right\}$ For the sake of contradiction, let S be nonempty. By the well-ordering principle, there exists a least element $l \in S$.
    Note that if $i=1$,  $\gcd(p, q^1) = \gcd(p, q) = 1$. Thus, $1 \notin S$ and $l>1$, and $l-1 \notin S$.
    \begin{align*}
        \gcd(p, q^{l-1}) &= 1 \\
        \gcd(p, q) &= 1
        \gcd(p, q^l) &= 1
    \end{align*}
    Thus, this is a contradiction and $S$ is empty. An analogous argument can be made since  $\gcd(p, q^j) = 1$ proving that $ \gcd(p^k, q^j) = 1$.
\end{proof}
