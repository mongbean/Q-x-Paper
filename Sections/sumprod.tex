\begin{lemma} $\sum\limits _{i=i_0}^{n}  a_i + \sum\limits _{i=i_0}^{n}  b_i = \sum\limits _{i=i_0}^{n} (a_i+b_i)$ \end{lemma}
\begin{proof}
Note that \[\sum_{i=i_0}^{i_0}  a_i+ \sum_{i=i_0}^{i_0}  b_i = a_{i_0}+b_{i_0}=\sum_{i=i_0}^{i_0}  (a_i+b_i)\]
Let $S=\left \{k \in \mathbb{Z}^+  \; \middle |  \;\sum\limits_{i=i_0}^{i_0+k}  a_i +  \sum\limits_{i=i_0}^{i_0+k}  b_i \neq  \sum\limits_{i=i_0}^{i_0+k} ( a_i +b_i)\right \}$. For the sake of contradiction, $S$ is not empty. As $S \subseteq \mathbb{Z}^+$, by the Well-Ordering Principle, there exists a least element $l \in S$. \\
If $k=1$: \[
 \sum_{i=i_0}^{i_0+1}  a_i+ \sum_{i=i_0}^{i_0+1}  b_i = a_{i_0+1}+ a_{i_0}+b_{i_0+1}+b_{i_0}= (a_{i_0+1}+b_{i_0+1})+(a_{i_0}+b_{i_0}) =\sum_{i=i_0}^{i_0+1}  (a_i+b_i)
\]
Thus, $1 \notin S$ and so $l>1$ by NIBZO. Since $l-1 \in \mathbb{Z}$ yet $l-1 \notin S$, 
\begin{align*} \sum\limits _{i=i_0}^{i_0+l-1}  a_i + \sum\limits _{i=i_0}^{i_0+l-1}  b_i &= \sum\limits _{i=i_0}^{i_0+l-1} (a_i+b_i) \\
a_l + b_l+\sum\limits _{i=i_0}^{i_0+l-1}  a_i + \sum\limits _{i=i_0}^{i_0+l-1}  b_i &= a_l + b_l+\sum\limits _{i=i_0}^{i_0+l-1} (a_i+b_i)\\
a_l + \sum\limits _{i=i_0}^{i_0+l-1}  a_i + b_l+\sum\limits _{i=i_0}^{i_0+l-1}  b_i &= (a_l + b_l)+\sum\limits _{i=i_0}^{i_0+l-1} (a_i+b_i)\\
\sum\limits _{i=i_0}^{i_0+l}  a_i + \sum\limits _{i=i_0}^{i_0+l}  b_i &= \sum\limits _{i=i_0}^{i_0+l} (a_i+b_i)\\
\end{align*}
However, since $l \in S$, this is a contradiction. Thus, S must be empty.\end{proof}

\begin{lemma}
Sum Constants: $\sum\limits _{i=i_0}^{n} ca_i = c\sum\limits _{i=i_0}^{n} a_i$\end{lemma}
\begin{proof} Note that \[
\sum\limits _{i=i_0}^{i_0} ca_i = ca_{i_0}=c\sum\limits _{i=i_0}^{i_0} a_i
\] 
and that 
\[
\sum\limits _{i=i_0}^{i_0+1} ca_i = ca_{i_0} + ca_{i_0+1}=c\sum\limits _{i=i_0}^{i_0+1} a_i
\] 
Let $S=\left \{k \in \mathbb{Z}^+  \;\middle| \;\sum\limits_{i=i_0}^{i_0+k}  ca_i \neq  c\sum\limits_{i=i_0}^{i_0+k} a_i\right \}$. For the sake of contradiction, $S$ is not empty. As $S \subseteq \mathbb{Z}^+$, by the well-ordering principle, there exists a least element $l \in S$. Since $1 \notin S$, $l>1$ by NIBZO. As $l-1 \in \mathbb{Z}$ yet $l-1 \notin S$, 
\begin{align*}
    \sum\limits_{i=i_0}^{i_0+l-1}  ca_i &=  c\sum\limits_{i=i_0}^{i_0+l-1} a_i \\
    ca_l + \sum\limits_{i=i_0}^{i_0+l-1}  ca_i &=  ca_l+c\sum\limits_{i=i_0}^{i_0+l-1} a_i \\
   \sum\limits_{i=i_0}^{i_0+l}  ca_i &=  c\sum\limits_{i=i_0}^{i_0+l} a_i 
\end{align*}
However, since $l \in S$, this is a contradiction. Thus, S must be empty.\end{proof}

\begin{lemma} Multiplied Sums: $\sum\limits _{i=i_0}^{n}  a_i \cdot \sum\limits _{j=j_0}^{m}  b_j = \sum\limits _{i=i_0}^{n} \sum\limits _{j=j_0}^{m} (a_ib_j)$ \end{lemma}
\begin{proof} By Sum Constants, we know that 
\begin{align*} \sum\limits _{i=i_0}^{n}  a_i \cdot \sum\limits _{j=j_0}^{m}  b_j &= \sum\limits _{i=i_0}^{n} (a_i \cdot \sum\limits _{j=j_0}^{m}  b_j) \\
 \sum\limits _{i=i_0}^{n}  a_i \cdot \sum\limits _{j=j_0}^{m}  b_j &= \sum\limits _{i=i_0}^{n} (\sum\limits _{j=j_0}^{m}  a_i b_j) 
\end{align*} \end{proof}


\begin{lemma} $\prod\limits _{i=i_0}^{n}  a_i  \prod\limits _{i=i_0}^{n}  b_i = \prod\limits _{i=i_0}^{n} a_ib_i$\end{lemma}
\begin{proof}
 \normalfont Note that \[\prod_{i=i_0}^{i_0}  a_i \prod_{i=i_0}^{i_0}  b_i = a_{i_0}b_{i_0}=\prod_{i=i_0}^{i_0}  a_ib_i\]
Let $S=\left \{k \in \mathbb{Z}^+  \; \middle |  \;\prod\limits_{i=i_0}^{i_0+k}  a_i   \prod\limits_{i=i_0}^{i_0+k}  b_i \neq  \prod\limits_{i=i_0}^{i_0+k}  a_i b_i\right \}$. For the sake of contradiction, $S$ is not empty. As $S \subseteq \mathbb{Z}^+$, by the Well-Ordering Principle, there exists a least element $l \in S$. \\
If $k=1$: \[
 \prod_{i=i_0}^{i_0+1}  a_i \prod_{i=i_0}^{i_0+1}  b_i = a_{i_0+1} a_{i_0}\cdot b_{i_0+1}b_{i_0}= (a_{i_0+1}b_{i_0+1})(a_{i_0}b_{i_0}) =\prod_{i=i_0}^{i_0+1}  a_ib_i
\]
Thus, $1 \notin S$ and so $l>1$ by NIBZO. Since $l-1 \in \mathbb{Z}$ yet $l-1 \notin S$, 
\begin{align*} \prod\limits _{i=i_0}^{i_0+l-1}  a_i  \prod\limits _{i=i_0}^{i_0+l-1}  b_i &= \prod\limits _{i=i_0}^{i_0+l-1} a_ib_i \\
a_l  b_l \cdot \prod\limits _{i=i_0}^{i_0+l-1}  a_i  \prod\limits _{i=i_0}^{i_0+l-1}  b_i &= a_l  b_l \cdot\prod\limits _{i=i_0}^{i_0+l-1} a_ib_i\\
(a_l  \prod\limits _{i=i_0}^{i_0+l-1}  a_i) \cdot (b_l \prod\limits _{i=i_0}^{i_0+l-1}  b_i) &= (a_l  b_l)\prod\limits _{i=i_0}^{i_0+l-1} a_ib_i\\
\prod\limits _{i=i_0}^{i_0+l}  a_i  \prod\limits _{i=i_0}^{i_0+l}  b_i &= \prod\limits _{i=i_0}^{i_0+l} a_ib_i\\
\end{align*}
However, since $l \in S$, this is a contradiction. Thus, S must be empty. \end{proof}

\begin{lemma} $\prod\limits _{i=i_0}^{n}  a_i = \prod\limits _{i=i_0}^{k-1} a_i \m a_k \m\prod\limits _{i=k+1}^{n} a_i$\end{lemma}
\begin{proof}
Let $S=\left\{t\in\mathbb{Z}^+ \;\middle|\; \prod\limits_{i=i_0}^{k+t} a_i \neq \prod\limits_{i=i_0}^{k-1} a_i \m a_k \m \prod\limits_{i=k+1}^{k+t} a_i\right\}$. We show $S$ is empty. First, $1\notin S$, since by the recursive definition of product,
\[
\prod\limits_{i=i_0}^{k+1}a_i
=
a_{k+1}\m\prod\limits_{i=i_0}^{k}a_i
=
a_{k+1}\m a_k\m\prod\limits_{i=i_0}^{k-1}a_i
=
\prod\limits_{i=i_0}^{k-1}a_i\m a_k\m
\prod\limits_{i=k+1}^{k+1}a_i.
\]
For the sake of contradiction, suppose $S$ is not empty. Since $S\subseteq\mathbb{Z}^+$, by WOP there is a least element $l\in S$. Since $1\notin S$, $l>1$, so $l-1\in\mathbb{Z}^+$. Since $l$ is least in $S$, $l-1\notin S$. Thus $\prod\limits_{i=i_0}^{k+l-1} a_i = \prod\limits_{i=i_0}^{k-1} a_i \m a_k \m \prod\limits_{i=k+1}^{k+l-1} a_i$. Then, using the recursive definition again,
\begin{align*}
\prod\limits_{i=i_0}^{k+l} a_i
&=a_{k+l}\m\prod\limits_{i=i_0}^{k+l-1}a_i\\
&=a_{k+l}\m
\left(\prod\limits_{i=i_0}^{k-1} a_i \m a_k \m
\prod\limits_{i=k+1}^{k+l-1} a_i\right)\\
&=\prod\limits_{i=i_0}^{k-1} a_i \m a_k \m
\prod\limits_{i=k+1}^{k+l}a_i.
\end{align*}
So $l\notin S$, a contradiction. Therefore $S$ is empty. Since $n>k$, write $n=k+t$ for some $t\in\mathbb{Z}^+$. Since $S$ is empty, the equality holds for $n$.
\end{proof}
